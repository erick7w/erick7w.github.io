Elasticsearch-curd操作02

// 查看节点
GET _cat/nodes
// 查看分片
GET _cat/shards

// create document. 自动生成id
POST users/_doc
{
“user”: “Mike”,
“post_da te”: “2022-11-01T22:24:56”,
“message”: “trying out kibana”
}

// create document. 指定id，如果id已存在，则报错
PUT users/_doc/1?op_type=create
{
“user”: “Jack”,
“post_date”: “2022-11-02T22:24:56”,
“message”: “trying out kibana”
}

// create document. 指定id，如果已经存在，则报错
PUT users/_create/1
{
“user”: “Lili”
“post_date”: “2022-11-02T22:24:56”,
“message”: “good girl”
}

// get document
GET users/_doc/1
// alter document
PUT users/_doc/1
{
“user”: “Mike”
}
// 在原文档上增加字段
POST users/_update/1
{
“doc”:{
“post_date”: “2022-11-02T22:24:50”,
“message”: “bad boy”
}
}
// delete 删除文档
DELETE users/_doc/1

// 批量操作可以在一个请求中对多个索引进行curd,部分失败不会影响其他的操作
// _bulk 批量操作，
POST _bulk
{ “index”: { “_index”: “test”, “_id”: “1”}}
{ “field1”: “value1”}
{ “delete”: {“_index”: “test”, “_id”: “2”}}
{ “create”: {“_index”: “test2”, “_id”: “3”}}
{ “field1”: “value3”}
{ “update”: {“_id”: “1”, “_index”: “test”}}
{ “doc”: { “field2”: “value2”}}

// _mget 批量读取
GET /_mget
{
“docs”: [
{
“_index”: “test”,
“_id”: “1”
},
{
“_index”: “test2”,
“_id”: “2”
}
]
}

// _msearch 批量搜索
POST Kibana_sample_data_ecommerce/_msearch
{}
{ “query”: {“match_all”: {}}, “size”:1 }
{ “index”: “kibana_sample_data_flights”}
{ “query”: {“match_all”: {}}, “size”: 2}
